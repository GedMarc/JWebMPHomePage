//You can create distinctly separate modules when running in a container, or performing migration.
//javax.inject is fully supported, you can clean split from EJB using the guice package instead
@com.google.inject.Singleton
public class HtmlTagsDemoScreen
  extends PluginDemoScreen
{
result = new ClassGraph()
 .whitelistPackages("com.jwebmp.core.base.html")
 .enableClassInfo()
 .scan();

public HtmlTagsDemoScreen()
{
super("HtmlTags", "Base", "HTML Tags");
    for (ClassInfo allClass : result.getAllClasses())
    {
        Class clazz = allClass.loadClass();
        String htmlTag = clazz.getConstructor()
              .newInstance()
              .setRenderIDAttribute(false)
              .toString(0);

        addComponentTile(clazz.getSimpleName(),
            StringEscapeUtils.escapeHtml4(htmlTag)
        );
    }
}
}
